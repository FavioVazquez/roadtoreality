# Plan 06-04: Stub Pages + stop-shell.js Updates

## Objective

Four coordinated changes that must land together: (1) update `stop-shell.js` with keyboard arrow navigation and `data-autoplay` support; (2) update `nav.js` and `components.css` simultaneously to make stub cards clickable (these two files are tightly coupled — changing one without the other creates a broken half-state); (3) build rich canvas teaser animations for all 38 stub stops, each unique to its physics concept; (4) upgrade the 38 stub page HTML files with the teaser script, full page content from stops.json description, and a GitHub contribution link.

## Context

**stop-shell.js arrow nav:** The new keydown listener must check two guards before acting:
1. `document.activeElement` is not an INPUT, TEXTAREA, or SELECT (sliders use arrow keys)
2. `document.getElementById('search-modal').hasAttribute('open')` is false (Plan 06-03 contract)

Navigation targets come from `config.prev` and `config.next` (already parsed by `_readConfig()`).

**data-autoplay:** Lives on `.sim-wrapper` (not `#sim-mount`). The IntersectionObserver callback reads it via `mount.closest('.sim-wrapper').dataset.autoplay`. When `data-autoplay="false"`, the observer fires but does NOT call `api.start()`. The observer still calls `api.pause()` on exit regardless.

**nav.js + components.css coupling:** These two files must be changed in the same plan:
- `nav.js _cardHtml()`: change `tag = 'div'` to `tag = 'a'`, change `href = '#'` to `href = 'stops/' + stop.id + '/index.html'`, remove `aria-hidden="true"` from stub card attrs.
- `components.css`: remove `opacity: 0.5` and `pointer-events: none` from `.stop-card--stub`. Convert `::after` text "Coming soon" to a styled pill badge.

**Stub teaser animations:** Each of the 38 stubs gets a canvas animation that is:
- Self-contained IIFE setting `window.SimAPI = { start, pause, reset, destroy }`
- Loaded via `<script src="sim.js"></script>` in the stub's HTML (same as implemented stops)
- Visually represents the physics concept (not a generic spinner)

- Uses vanilla Canvas 2D (p5.js is available at `../../assets/js/vendor/p5.min.js` if needed)
- Under ~120 lines of canvas code for teaser quality (not a full simulation)

**Stub page content upgrade:** Each stub's HTML is currently minimal (one paragraph). Replace with:
- Full intro paragraph from stops.json `description` field (already in the file in most cases — check and expand if too short)
- A "What you'll explore" 2–3 item list hinting at what the full simulation will show
- The canvas teaser in `<div class="sim-wrapper" data-autoplay="true">` wrapping `<div class="sim-container" id="sim-mount">`

- A GitHub contribution link: `<a href="https://github.com/FavioVazquez/roadtoreality/issues" target="_blank" rel="noopener">Contribute the full simulation on GitHub</a>`

```javascript
function _bindArrowNav(cfg) {
  if (!cfg) return;
  document.addEventListener('keydown', function (e) {
    /* Guard 1: ignore when an input/select/textarea is focused */
    var active = document.activeElement;
    if (active && (
      active.tagName === 'INPUT' ||
      active.tagName === 'TEXTAREA' ||
      active.tagName === 'SELECT'
    )) return;

    /* Guard 2: ignore when search modal is open */
    var modal = document.getElementById('search-modal');
    if (modal && modal.hasAttribute('open')) return;

    if (e.key === 'ArrowLeft' && cfg.prev) {
      window.location.href = '../' + cfg.prev + '/index.html';
    }
    if (e.key === 'ArrowRight' && cfg.next) {
      window.location.href = '../' + cfg.next + '/index.html';
    }
  });
}
```

**Modify `_initSimObserver()` to read `data-autoplay`:**

Replace the existing `_initSimObserver` function with this version:

```javascript
function _initSimObserver() {
  var mount = document.getElementById('sim-mount');
  if (!mount) return;

  var observer = new IntersectionObserver(function (entries) {
    entries.forEach(function (entry) {
      var api = window.SimAPI;
      if (!api) return;

      var wrapper = mount.closest('.sim-wrapper');
      var autoplay = wrapper ? wrapper.dataset.autoplay !== 'false' : true;

      if (entry.isIntersecting) {
        if (autoplay && typeof api.start === 'function') api.start();
        var dot = document.querySelector('.sim-caption__dot');
        if (dot && autoplay) dot.classList.add('is-running');
      } else {
        if (typeof api.pause === 'function') api.pause();
        var dot = document.querySelector('.sim-caption__dot');
        if (dot) dot.classList.remove('is-running');
      }
    });
  }, { threshold: 0.15 });

  observer.observe(mount);
}
```

**SimAPI contract for every sim.js:**

```javascript
(function () {
  'use strict';
  var mount = document.getElementById('sim-mount');
  if (!mount) return;

  var canvas = document.createElement('canvas');
  canvas.width  = mount.clientWidth  || 600;
  canvas.height = mount.clientHeight || 360;
  canvas.style.display = 'block';
  canvas.style.width  = '100%';
  canvas.style.height = '100%';
  mount.appendChild(canvas);
  var ctx = canvas.getContext('2d');

  var raf = null;
  var t = 0;
  var running = false;

  var reduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  function drawFrame() {
    t += 0.016; /* ~60fps time step */
    /* ... concept-specific drawing code ... */
    if (running && !reduced) {
      raf = requestAnimationFrame(drawFrame);
    }
  }

  function drawStatic() {
    t = 0;
    /* ... render one frame at t=0 ... */
  }

  window.SimAPI = {
    start: function () {
      if (running) return;
      running = true;
      if (reduced) { drawStatic(); return; }
      raf = requestAnimationFrame(drawFrame);
    },
    pause: function () {
      running = false;
      if (raf) { cancelAnimationFrame(raf); raf = null; }

    }
  };

  /* Draw initial static frame so the canvas is not blank before play */

}());
```

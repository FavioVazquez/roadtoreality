Field line algorithm (run inside `draw()` every frame):
```
function traceFieldLines() {
  var nRays = 8;
  for each charge c where c.q > 0:
    for i = 0 to nRays-1:
      var angle = i * 2 * Math.PI / nRays;
      var px = c.x + 12 * Math.cos(angle);
      var py = c.y + 12 * Math.sin(angle);
      ctx.beginPath(); ctx.moveTo(px, py);
      for step = 0 to 499:
        var Ex = 0, Ey = 0;
        for each charge cc:
          var dx = px - cc.x; var dy = py - cc.y;
          var r2 = dx*dx + dy*dy; var r = Math.sqrt(r2);
          if (r < 0.1) break inner;
          Ex += cc.q * dx / (r2 * r);
          Ey += cc.q * dy / (r2 * r);
        var emag = Math.sqrt(Ex*Ex + Ey*Ey);
        if (emag < 1e-8) break;
        px += (Ex/emag) * 3; py += (Ey/emag) * 3;
        ctx.lineTo(px, py);
        // Termination:
        if (px < 0 || px > splitX || py < 0 || py > H) break;
        for each charge cc:
          if (dist(px,py, cc.x,cc.y) < 10) break outer;
      ctx.strokeStyle = 'rgba(82,133,200,0.45)'; ctx.lineWidth = 1; ctx.stroke();
}
```

Wire in `setup()`:
```
canvas.addEventListener('mousedown', function(e) { startDrag(e); });
canvas.addEventListener('mousemove', function(e) { doDrag(e); });
canvas.addEventListener('mouseup', function(e) { endDrag(e); });
canvas.addEventListener('touchstart', function(e) { e.preventDefault(); startDrag(e); }, {passive:false});
canvas.addEventListener('touchmove', function(e) { e.preventDefault(); doDrag(e); }, {passive:false});
canvas.addEventListener('touchend', function(e) { endDrag(e); });

function getCanvasPos(e) {
  var pos = e.touches ? e.touches[0] : e;
  var rect = canvas.getBoundingClientRect();
  return { x: pos.clientX - rect.left, y: pos.clientY - rect.top };
}
function startDrag(e) {
  var p = getCanvasPos(e);
  for (var i = 0; i < charges.length; i++) {
    var dx = p.x - charges[i].x; var dy = p.y - charges[i].y;
    if (Math.sqrt(dx*dx + dy*dy) < 20) { charges[i].dragging = true; break; }
  }
}
function doDrag(e) {
  var p = getCanvasPos(e);
  for (var i = 0; i < charges.length; i++) {
    if (charges[i].dragging) {
      charges[i].x = Math.max(14, Math.min(splitX - 14, p.x));
      charges[i].y = Math.max(14, Math.min(H - 14, p.y));
    }
  }
}
function endDrag(e) {
  for (var i = 0; i < charges.length; i++) { charges[i].dragging = false; }
}
```

function resize() {
  var rect = mount.getBoundingClientRect();

  W = Math.round(rect.width || 600);
